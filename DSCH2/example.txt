module example( clk2,clk1,out1);
 input clk2,clk1;
 output out1;
 xor #(16) xor2(out1,clk1,clk2);
endmodule

always
#2000 clk2=~clk2;
#1000 clk1=~clk1;

// clk1 CLK 10.000 10.000
